import smtplib
from email.mime.text import MIMEText

#############

#要发给谁，这里发给2个人
mailto_list=[“a@qq.com”,”b@qq.com”]

#设置服务器，用户名、口令以及邮箱的后缀
mail_host=”smtp.126.com”

######################
def send_mail(to_list,sub,content):
