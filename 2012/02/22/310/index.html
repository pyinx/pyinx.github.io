def num(x):
    if x<1:
        meiyuan = x * 100
        _25 = meiyuan / 25
        _251 = meiyuan % 25
        _10 = _251 / 10
        _101 = _251 % 10
        _5 = _101 / 5
        _51 = _101 % 5
        _1=_51 / 1
        print  "25美分换算和 " , _25,""
        print  "10美分换算和 " , _10,""
        print  "5美分换算和 " , _5,""
        print  "1美分换算和 " , _1,""

        print "输入小于1的数值"
