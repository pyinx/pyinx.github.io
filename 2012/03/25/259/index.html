#!/usr/bin/env python

import sys,urllib2

url=’http://www.baidu.com‘

req=urllib2.Request(url)

fd=urllib2.urlopen(req)

f=open(r’c:\test.html’,’a’)

while 1:

data=fd.read(2048)

if not data:

    f.close()

    break
